static Response standardJsonResponse(
    {Map? body, Map<String, String> headers = const {}}) {
  return Response.ok(
    body != null ? JsonEncoder().convert(body) : '{}',
    headers: {
      ...commonHeaders,
      ...{
        'Cache-Control': 'none',
      },
      ...headers,
    },
  );
}
